CREATE VIEW VistaAlumnosChiclana AS
SELECT nombre, fecha_nacimiento
FROM Alumnos
WHERE ciudad = 'Chiclana'
WITH READ ONLY; 

--2
SELECT VIEW_NAME, TEXT
FROM USER_VIEWS
WHERE VIEW_NAME = 'VISTAALUMNOSCHICLANA';

--3
CREATE INDEX idx_alumnos_fecha_nacimiento ON Alumnos (fecha_nacimiento);

--4
SELECT INDEX_NAME, TABLE_NAME
FROM USER_INDEXES
WHERE TABLE_NAME = 'ALUMNOS';

--5
CREATE PROFILE MiPerfil LIMIT
IDLE_TIME 20;

--6
CREATE TABLESPACE TSUsuario DATAFILE 'TSUsuario01.dbf' 
SIZE 100M AUTOEXTEND ON NEXT 10M MAXSIZE UNLIMITED;

CREATE PROFILE PERFIL_SEGURIDAD_P6 LIMIT
    PASSWORD_LIFE_TIME 14                
    FAILED_LOGIN_ATTEMPTS 1             
    PASSWORD_LOCK_TIME 10;   

CREATE ROLE ROL_ACTIVIDADES_ADICIONALES;
GRANT
    CREATE ANY VIEW,     
    GRANT ANY OBJECT PRIVILEGE 
TO ROL_ACTIVIDADES_ADICIONALES;

-- Conceder INSERT y UPDATE en HR.EMPLOYEES
-- Asumimos que HR es el esquema propietario de EMPLOYEES
GRANT INSERT, UPDATE ON HR.EMPLOYEES TO ROL_ACTIVIDADES_ADICIONALES; 
    
CREATE USER USRPRACTICA1 IDENTIFIED BY "ClaveSeguraP1!"
DEFAULT TABLESPACE TSUsuario
TEMPORARY TABLESPACE TEMP
PROFILE PERFIL_SEGURIDAD_P6;

CREATE USER USRPRACTICA2 IDENTIFIED BY "ClaveSeguraP2!"
DEFAULT TABLESPACE TSUsuario
TEMPORARY TABLESPACE TEMP
PROFILE PERFIL_SEGURIDAD_P6;

ALTER USER USRPRACTICA1 QUOTA 150M ON TSUsuario;

 -- Otorgar CREATE SESSION a ambos usuarios
GRANT CREATE SESSION TO USRPRACTICA1, USRPRACTICA2;

Otorgar el rol de actividades adicionales
GRANT ROL_ACTIVIDADES_ADICIONALES TO USRPRACTICA1, USRPRACTICA2;

SELECT USERNAME, PROFILE

--8
SELECT GRANTEE, GRANTED_ROLE
FROM DBA_ROLE_PRIVS
WHERE GRANTED_ROLE = 'ROL_ACTIVIDADES_ADICIONALES';
