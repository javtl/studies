CREATE TABLE Clientes (
	dni VARCHAR2(9) PRIMARY KEY,
	nombre VARCHAR2(50),
	apellidos VARCHAR2(100),
	edad NUMBER
);
CREATE TABLE Coches(
	codigo NUMBER PRIMARY KEY,
	matricula VARCHAR2(10) UNIQUE,
	marca VARCHAR2(50),
	modelo VARCHAR2(50),
	color VARCHAR2(30),
	dni_cliente VARCHAR2(9),
	CONSTRAINT fk_dni_cliente FOREIGN KEY (dni_cliente) REFERENCES Clientes(dni)
);
