FROM ASIGNATURAS

CREATE OR REPLACE VIEW AL_2022 AS
SELECT Nombre, Apellidos, Direccion, CP
FROM ALUMNOS
WHERE CP = 11004
WITH READ ONLY;

--4
SELECT view_name, text
FROM user_views
WHERE view_name = 'AL_2022'
